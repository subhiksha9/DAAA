#include<stdio.h>
#include<string.h>
int lcs(char c1[],char c2[],int m,int n)
{
    int a[n+1][m+1];
    for(int i=0;i<m+1;i++)
      a[0][i]=0;
    for(int i=0;i<n+1;i++)
       a[i][0]=0;
    for (int i=0;i<n;i++)
    {
        for (int j=0;j<m;j++)
        {
            if(c2[i]==c1[j])
            a[i+1][j+1]=a[i][j]+1;
            else
             a[i+1][j+1]=a[i+1][j]>a[i][j+1]?a[i+1][j]:a[i][j+1];
        }
    }
    char sq[a[n][m]];
    int c=0,i=n,j=m;
    while(i!=0 && j!=0)
    {
        if(a[i][j]==a[i][j-1])
         j=j-1;
        else if(a[i][j]==a[i-1][j])
         i=i-1;
        else
          {
              sq[c]=c2[i-1];
              c++;
              i=(i!=0)?i-1:0;
              j=(j!=0)?j-1:0;
          }
    }
   // printf("%s",(sq));
   printf("Longest subsequence: ");
    for(int i=c-1;i>=0;i--)
      printf("%c",sq[i]);
    printf("\nLength:%d",a[n][m]);
}
void main()
{
    char c1[]="AGGTAB";
    char c2[]="GXTXAYB";
    int m=strlen(c1),n=strlen(c2);
    lcs(c1,c2,m,n);
}
